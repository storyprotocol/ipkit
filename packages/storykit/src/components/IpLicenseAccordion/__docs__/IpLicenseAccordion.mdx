# IpLicenseAccordion

Displays the IP Asset Licenses in a readable format (cans & cannots).

#### Example

<Canvas of={IpLicenseAccordionStory.Select} />

## Usage

```ts
import { IpProvider } from "@storyprotocol/storykit"
import { IpLicenseAccordion } from "@storyprotocol/storykit"

const ExamplePage = () => {
  return (
    <IpProvider ipId={"0xbbf08a30b9ff0f717a024a75963d3196aaf0f0dd"}  options={{ licenseTermsData: true }}>
      <IpLicenseAccordion />
    </IpProvider>
  );
};

export default ExamplePage;
```
